## エンジニア視点の意見

**実現可能性：3案とも単一HTMLモックで十分作れます。** レーダーチャート・絵文字スライダー・診断ロジックすべてCSS+JSで実装可能。API連携部分はダミーデータで再現し、本実装時に差し替える形で問題なし。

**開発難易度：**
- **ワクワクマップ**（レーダーチャート）：普通。Canvas or SVGで描画、リアルタイム更新もJS内で完結する
- **ワクワクログ**（絵文字スライダー）：軽い。input range＋絵文字マッピングで素直に実装できる
- **ワクメモ**（3択＋診断）：軽い。最もシンプルで、記録2〜3件でも診断が破綻しにくい構造

**PM補佐の指摘を踏まえた技術的意見：**
- **「記録2〜3件で成立するか」が最重要。** レーダーチャート（ワクワクマップ）は記録が少ないとスカスカで寂しい。3択タップの累積（ワクメモ方式）なら2件でもタイプ傾向は出せる。ここは設計上の差が大きい
- **親のスマホ前提** → LocalStorageで状態保持すれば、親が途中でブラウザを閉じても復帰可能。ただし「兄弟で1台を使い回す」ケースを考えると、**起動時に「だれが使う？」の名前選択**を1ステップ入れるべき。軽い実装で対応できる
- **バス車内の活用** → 技術的に面白い案として、**移動中に「さっきの体験どうだった？」のプッシュ的バナー表示**をタイマーで再現できる。実装は軽い（setTimeout＋モーダル）

**結論：ワクメモベースが最も堅実。** 記録少数でも成立し、実装が軽く、親子利用の拡張もしやすい。ワクワクマップのレーダーチャートは「育てる楽しさ」があるので、ワクメモの診断結果画面にビジュアル要素として組み込むハイブリッドが最善と考えます。

function drawRadar(svgEl, scores, size, opts = {}) {
  const { showLabels = false, showValues = false, animate = false } = opts;
  const cx = size / 2, cy = size / 2, r = size * 0.35;
  const n = AXES.length;
  let svg = '';

  // 背景グリッド
  for (let lv = 1; lv <= 3; lv++) {
    const rr = r * lv / 3;
    const pts = [];
    for (let i = 0; i < n; i++) {
      const angle = (Math.PI * 2 * i / n) - Math.PI / 2;
      pts.push(`${cx + rr * Math.cos(angle)},${cy + rr * Math.sin(angle)}`);
    }
    svg += `<polygon points="${pts.join(' ')}" fill="none" stroke="#E0E0E0" stroke-width="${lv===3?1.5:0.7}"/>`;
  }

  // 軸線
  for (let i = 0; i < n; i++) {
    const angle = (Math.PI * 2 * i / n) - Math.PI / 2;
    const x2 = cx + r * Math.cos(angle);
    const y2 = cy + r * Math.sin(angle);
    svg += `<line x1="${cx}" y1="${cy}" x2="${x2}" y2="${y2}" stroke="#E0E0E0" stroke-width="0.7"/>`;
  }

  // データ領域
  const dataPts = [];
  for (let i = 0; i < n; i++) {
    const val = Math.min(scores[i] || 0, 1);
    const angle = (Math.PI * 2 * i / n) - Math.PI / 2;
    const dr = r * (0.1 + 0.9 * val);
    dataPts.push(`${cx + dr * Math.cos(angle)},${cy + dr * Math.sin(angle)}`);
  }
  const hasData = scores.some(s => s > 0);
  if (hasData) {
    svg += `<polygon points="${dataPts.join(' ')}" fill="rgba(255,107,53,0.15)" stroke="var(--primary)" stroke-width="2" stroke-linejoin="round"${animate ? ' style="animation:fadeUp 0.5s ease"' : ''}/>`;
    // 頂点のドット
    for (let i = 0; i < n; i++) {
      if ((scores[i] || 0) > 0) {
        svg += `<circle cx="${dataPts[i].split(',')[0]}" cy="${dataPts[i].split(',')[1]}" r="3.5" fill="var(--primary)"/>`;
      }
    }
  }

  // ラベル
  if (showLabels) {
    for (let i = 0; i < n; i++) {
      const angle = (Math.PI * 2 * i / n) - Math.PI / 2;
      const lr = r + size * 0.12;
      const lx = cx + lr * Math.cos(angle);
      const ly = cy + lr * Math.sin(angle);
      const val = scores[i] || 0;
      const color = val > 0 ? AXIS_COLORS[i] : '#AAA';
      const fontSize = size > 100 ? 12 : 6;
      svg += `<text x="${lx}" y="${ly}" text-anchor="middle" dominant-baseline="central" fill="${color}" font-size="${fontSize}" font-weight="700">${AXES[i]}</text>`;
      if (showValues && val === 0) {
        const vly = ly + fontSize + 2;
        svg += `<text x="${lx}" y="${vly}" text-anchor="middle" fill="#CCC" font-size="${fontSize * 0.7}">未冒険</text>`;
      }
    }
  }

  svgEl.innerHTML = svg;
}

/* ===== ユーザーのスコア計算 ===== */
function calcScores(user) {
  const scores = [0,0,0,0,0,0];
  const records = user.records;
  const count = Object.keys(records).length;
  if (count === 0) return scores;

  for (const boothId in records) {
    const booth = BOOTHS.find(b => b.id === parseInt(boothId));
    if (!booth) continue;
    const rec = records[boothId];
    // 感情によるウェイト
    let weight = 1;
    if (rec.q1 === 'fun') weight = 1.0;
    else if (rec.q1 === 'wow') weight = 1.2;
    else if (rec.q1 === 'curious') weight = 1.1;
    // 難易度ボーナス
    let diffBonus = 0;
    if (rec.q2 === 'hard') diffBonus = 0.15;
    else if (rec.q2 === 'normal') diffBonus = 0.05;

    for (let i = 0; i < 6; i++) {
      scores[i] += booth.axes[i] * weight + (booth.axes[i] > 0 ? diffBonus : 0);
    }
  }

  // 正規化（最大1）
  const max = Math.max(...scores, 0.001);
  return scores.map(s => Math.min(s / max, 1));
}

function renderStep() {
  // ステップインジケータ
  const indicator = document.getElementById('stepIndicator');

  for (let i = 0; i < QUESTIONS.length; i++) {
    const dot = document.createElement('div');
    dot.className = 'step-dot' + (i === currentStep ? ' active' : (i < currentStep ? ' done' : ''));
    indicator.appendChild(dot);
  }

  const area = document.getElementById('questionArea');
  const chartPreview = document.getElementById('recordChartPreview');

  if (currentStep >= QUESTIONS.length) {
    // 完了表示
    area.innerHTML = '';
    const doneDiv = document.createElement('div');
    doneDiv.className = 'record-done-msg';
    doneDiv.innerHTML = `<h2>きろくできた！ 🎉</h2><p>マップが少し育ったよ。<br>ほかのブースも体験してみよう！</p>`;
    area.appendChild(doneDiv);

    const user = getCurrentUser(appState);
    const scores = calcScores(user);
    chartPreview.style.display = 'flex';
    drawRadar(document.getElementById('recordChart'), scores, 200, { showLabels: true, animate: true });

    setTimeout(() => {
      showScreen('listScreen');
      renderList();
    }, 2000);
    return;
  }

  chartPreview.style.display = 'none';
  const q = QUESTIONS[currentStep];
  area.innerHTML = `<div class="question-text">${q.text}</div>`;
  const choicesDiv = document.createElement('div');
  choicesDiv.className = 'choices';
  q.choices.forEach(c => {
    const btn = document.createElement('button');
    btn.className = 'choice-btn';
    const prevAnswer = recordAnswers[`q${currentStep + 1}`];
    if (prevAnswer === c.value) btn.classList.add('selected');
    btn.innerHTML = `<span class="choice-emoji">${c.emoji}</span>${c.label}`;
    btn.addEventListener('click', () => {
      recordAnswers[`q${currentStep + 1}`] = c.value;
      // 保存
      const user = getCurrentUser(appState);
      user.records[currentBooth.id] = { ...recordAnswers };
      saveState(appState);
      // 次へ
      currentStep++;
      setTimeout(renderStep, 200);
    });
    choicesDiv.appendChild(btn);
  });
  area.appendChild(choicesDiv);
}

    renderStep();
  } else {

    renderList();
  }
});

function renderReview() {
  const user = getCurrentUser(appState);
  const scores = calcScores(user);

  document.getElementById('reviewUserName').textContent = `${user.name}のワクワクマップ`;
  drawRadar(document.getElementById('reviewChart'), scores, 260, { showLabels: true, showValues: true, animate: true });

  // タイプ判定
  let maxIdx = 0;
  for (let i = 1; i < scores.length; i++) {
    if (scores[i] > scores[maxIdx]) maxIdx = i;
  }
  const axisName = AXES[maxIdx];
  const type = TYPES[axisName];
  document.getElementById('typeName').textContent = `${axisName}タイプ — ${type.name}`;
  document.getElementById('typeDesc').textContent = type.desc;

  // おすすめブース（未体験で相性良い上位2つ）
  const unvisited = BOOTHS.filter(b => !user.records[b.id]);
  const ranked = unvisited.map(b => {
    let affinity = 0;
    for (let i = 0; i < 6; i++) affinity += b.axes[i] * (1 - (scores[i] || 0));
    return { booth: b, affinity };
  }).sort((a, b) => b.affinity - a.affinity);

  const recommendList = document.getElementById('recommendList');
  const recommendSection = document.getElementById('recommendSection');
  if (ranked.length === 0) {
    recommendSection.style.display = 'none';
  } else {
    recommendSection.style.display = 'block';
    recommendList.innerHTML = '';
    ranked.slice(0, 2).forEach(r => {
      // 未冒険軸のうち、このブースが強い軸を理由に
      let reasonAxis = '';
      let maxVal = 0;
      for (let i = 0; i < 6; i++) {
        if (r.booth.axes[i] > maxVal && scores[i] < 0.3) {
          maxVal = r.booth.axes[i];
          reasonAxis = AXES[i];
        }
      }
      const reason = reasonAxis ? `「${reasonAxis}」の新しい発見があるかも！` : 'まだ体験していない分野だよ！';
      const card = document.createElement('div');
      card.className = 'recommend-card';

        <div class="booth-icon" style="background:${r.booth.color}">${r.booth.icon}</div>
        <div class="recommend-card-info">
          <div class="recommend-card-name">${r.booth.name}</div>
          <div class="recommend-card-reason">${reason}</div>
        </div>
      `;
      recommendList.appendChild(card);
    });
  }

  // 日常アクション
  const actionList = document.getElementById('actionList');
  actionList.innerHTML = '';
  type.actions.forEach((action, i) => {
    const card = document.createElement('div');
    card.className = 'action-card';
    card.innerHTML = `<div class="action-num">${i + 1}</div><div class="action-text">${action}</div>`;
    actionList.appendChild(card);
  });
}

3. **名前入力を後回しにする**（オンボーディング）→ 採用推奨。ただし兄弟切替との整合性をエンジニアに確認すべき（名前なしユーザーが2人同時に使い始めるケースの扱い）

**ふりかえり画面のスワイプカード分割**（ビジュアル提案）と**記録完了後の自動遷移廃止**（モバイル提案）も妥当だが、優先度は上記3点より低い。

**結論：v2で上記3点を修正し、完成判定して良い。** これ以上のサイクルは過剰改善のリスクがある。

function drawRadar(svgEl, scores, size, opts = {}) {
  const { showLabels = false, showValues = false, animate = false } = opts;
  const cx = size / 2, cy = size / 2, r = size * 0.35;
  const n = AXES.length;
  let svg = '';

  for (let lv = 1; lv <= 3; lv++) {
    const rr = r * lv / 3;
    const pts = [];
    for (let i = 0; i < n; i++) {
      const angle = (Math.PI * 2 * i / n) - Math.PI / 2;
      pts.push(`${cx + rr * Math.cos(angle)},${cy + rr * Math.sin(angle)}`);
    }
    svg += `<polygon points="${pts.join(' ')}" fill="none" stroke="#E0E0E0" stroke-width="${lv===3?1.5:0.7}"/>`;
  }

  for (let i = 0; i < n; i++) {
    const angle = (Math.PI * 2 * i / n) - Math.PI / 2;
    const x2 = cx + r * Math.cos(angle);
    const y2 = cy + r * Math.sin(angle);
    svg += `<line x1="${cx}" y1="${cy}" x2="${x2}" y2="${y2}" stroke="#E0E0E0" stroke-width="0.7"/>`;
  }

  const dataPts = [];
  for (let i = 0; i < n; i++) {
    const val = Math.min(scores[i] || 0, 1);
    const angle = (Math.PI * 2 * i / n) - Math.PI / 2;
    const dr = r * (0.1 + 0.9 * val);
    dataPts.push(`${cx + dr * Math.cos(angle)},${cy + dr * Math.sin(angle)}`);
  }
  const hasData = scores.some(s => s > 0);
  if (hasData) {
    svg += `<polygon points="${dataPts.join(' ')}" fill="rgba(255,107,53,0.15)" stroke="var(--primary)" stroke-width="2" stroke-linejoin="round"${animate ? ' style="animation:fadeUp 0.5s ease"' : ''}/>`;
    for (let i = 0; i < n; i++) {
      if ((scores[i] || 0) > 0) {
        svg += `<circle cx="${dataPts[i].split(',')[0]}" cy="${dataPts[i].split(',')[1]}" r="3.5" fill="var(--primary)"/>`;
      }
    }
  }

  if (showLabels) {
    for (let i = 0; i < n; i++) {
      const angle = (Math.PI * 2 * i / n) - Math.PI / 2;
      const lr = r + size * 0.12;
      const lx = cx + lr * Math.cos(angle);
      const ly = cy + lr * Math.sin(angle);
      const val = scores[i] || 0;
      const color = val > 0 ? AXIS_COLORS[i] : '#AAA';
      const fontSize = size > 100 ? 12 : 6;
      svg += `<text x="${lx}" y="${ly}" text-anchor="middle" dominant-baseline="central" fill="${color}" font-size="${fontSize}" font-weight="700">${AXES[i]}</text>`;
      if (showValues && val === 0) {
        const vly = ly + fontSize + 2;
        svg += `<text x="${lx}" y="${vly}" text-anchor="middle" fill="#CCC" font-size="${fontSize * 0.7}">未冒険</text>`;
      }
    }
  }

  svgEl.innerHTML = svg;
}

/* ===== ユーザーのスコア計算 ===== */
function calcScores(user) {
  const scores = [0,0,0,0,0,0];
  const records = user.records;
  const count = Object.keys(records).length;
  if (count === 0) return scores;

  for (const boothId in records) {
    const booth = BOOTHS.find(b => b.id === parseInt(boothId));
    if (!booth) continue;
    const rec = records[boothId];
    let weight = 1;
    if (rec.q1 === 'fun') weight = 1.0;
    else if (rec.q1 === 'wow') weight = 1.2;
    else if (rec.q1 === 'curious') weight = 1.1;
    let diffBonus = 0;
    if (rec.q2 === 'hard') diffBonus = 0.15;
    else if (rec.q2 === 'normal') diffBonus = 0.05;

    for (let i = 0; i < 6; i++) {
      scores[i] += booth.axes[i] * weight + (booth.axes[i] > 0 ? diffBonus : 0);
    }
  }

  const max = Math.max(...scores, 0.001);
  return scores.map(s => Math.min(s / max, 1));
}

    indicator.appendChild(dot);
  }

  const area = document.getElementById('questionArea');
  const chartPreview = document.getElementById('recordChartPreview');

  if (currentStep >= QUESTIONS.length) {

    drawRadar(document.getElementById('recordChart'), scores, 200, { showLabels: true, animate: true });
    return;

  chartPreview.style.display = 'none';
  const q = QUESTIONS[currentStep];
  area.innerHTML = `<div class="question-text">${q.text}</div>`;
  const choicesDiv = document.createElement('div');
  choicesDiv.className = 'choices';
  q.choices.forEach(c => {
    const btn = document.createElement('button');
    btn.className = 'choice-btn';
    const prevAnswer = recordAnswers[`q${currentStep + 1}`];
    if (prevAnswer === c.value) btn.classList.add('selected');
    btn.innerHTML = `<span class="choice-emoji">${c.emoji}</span><span class="choice-label">${c.label}</span>`;

      saveState(appState);
      currentStep++;
      setTimeout(renderStep, 200);
    });
    choicesDiv.appendChild(btn);
  });

function renderReview() {

  document.getElementById('reviewUserName').textContent = `${displayName}のワクワクマップ`;

  let maxIdx = 0;
  for (let i = 1; i < scores.length; i++) {
    if (scores[i] > scores[maxIdx]) maxIdx = i;
  }
  const axisName = AXES[maxIdx];
  const type = TYPES[axisName];
  document.getElementById('typeName').textContent = `${axisName}タイプ — ${type.name}`;
  document.getElementById('typeDesc').textContent = type.desc;

  const unvisited = BOOTHS.filter(b => !user.records[b.id]);
  const ranked = unvisited.map(b => {
    let affinity = 0;
    for (let i = 0; i < 6; i++) affinity += b.axes[i] * (1 - (scores[i] || 0));
    return { booth: b, affinity };
  }).sort((a, b) => b.affinity - a.affinity);

  const recommendList = document.getElementById('recommendList');
  const recommendSection = document.getElementById('recommendSection');
  if (ranked.length === 0) {
    recommendSection.style.display = 'none';
  } else {
    recommendSection.style.display = 'block';
    recommendList.innerHTML = '';
    ranked.slice(0, 2).forEach(r => {
      let reasonAxis = '';
      let maxVal = 0;
      for (let i = 0; i < 6; i++) {
        if (r.booth.axes[i] > maxVal && scores[i] < 0.3) {
          maxVal = r.booth.axes[i];
          reasonAxis = AXES[i];
        }
      }
      const reason = reasonAxis ? `「${reasonAxis}」の新しい発見があるかも！` : 'まだ体験していない分野だよ！';
      const card = document.createElement('div');
      card.className = 'recommend-card';
      card.innerHTML = `
        <div class="booth-icon" style="background:${r.booth.color}">${r.booth.icon}</div>
        <div class="recommend-card-info">
          <div class="recommend-card-name">${r.booth.name}</div>
          <div class="recommend-card-reason">${reason}</div>
        </div>
      `;
      recommendList.appendChild(card);
    });
  }

  const actionList = document.getElementById('actionList');
  actionList.innerHTML = '';
  type.actions.forEach((action, i) => {
    const card = document.createElement('div');
    card.className = 'action-card';
    card.innerHTML = `<div class="action-num">${i + 1}</div><div class="action-text">${action}</div>`;
    actionList.appendChild(card);
  });
}

## PM評価：ネクストアクションの質

**結論：現v2のネクストアクションは「正しく価値のある情報」を出せていない。改善必須。**

---

**問題の整理（3点）：**

### 1. 日常アクションが体験を反映していない
現状：`TYPES`オブジェクトにタイプ別の固定文言3つ（例：つくるタイプ → 「工作してみよう」「料理のお手伝い」「観察しよう」）。トラックづくりで「すごかった！」と答えてもパンづくりで「たのしかった！」と答えても同じ文言が出る。**入力に対して出力が無反応 = 抽象化が機能していない。**

### 2. おすすめブースに「観点」がない
現状：「まだ体験していない×未冒険軸が強い」ブースを出すが、理由が`「〇〇の新しい発見があるかも！`の1パターン。**「なぜそこに行くべきか」「何の視点で見るべきか」が欠落**。お題の「どの観点で回るといいか」に応えていない。

### 3. 体験の抽象化メッセージがない
現状：タイプ名＋固定説明文のみ。「きみは手を動かす体験で"すごい！"と感じやすいタイプだね」のような、**体験パターンからの個別化された気づき**がない。

---

## エンジニアへの実装指示

**v2のふりかえり画面を以下の3点で改修。画面構成・UIは変更しない。**

### 1. ブースデータに`afterActions`を追加
各ブースに「この体験後に日常で試せること」を感情別に持たせる。

```
afterActions: {
  fun: '家にある材料でミニカーを作ってみよう',
  wow: 'トラックの仕組みを図鑑で調べてみよう', 
  curious: '近所を走るトラックの種類を数えてみよう'
}
```

ふりかえり画面の「明日からやってみよう」は、**実際に記録したブース×感情回答の組み合わせ**から選出する（固定文言を廃止）。

### 2. おすすめブースに「観点」を付与
スコアの偏りから「きみは〇〇の体験が多かったから、次は△△の視点で見てみよう」を生成。具体例：

- 「つくる体験が多かったから、"まもる"の視点で消防士体験を見てみよう — 作ったものを守る側の仕事だよ」
- 軸間の関係性データ（`AXIS_RELATIONS`）を追加し、「なぜこの軸を次に試すべきか」の理由文を生成

### 3. 体験パターンの抽象化メッセージを追加
記録された感情回答×ブース軸の組み合わせから傾向文を生成。例：

- fun×つくる系2件 → 「きみは手を動かしてモノを作る体験がいちばん楽しいと感じるタイプ！」
- wow×しらべる系 → 「知らない世界を見たとき"すごい！"って思える好奇心の持ち主！」

タイプカードの説明文（`type.desc`固定文）をこの動的メッセージに差し替える。

/* ===== レーダーチャート描画 ===== */
function drawRadar(svgEl, scores, size, opts = {}) {
  const { showLabels = false, showValues = false, animate = false } = opts;
  const cx = size / 2, cy = size / 2, r = size * 0.35;
  const n = AXES.length;
  let svg = '';

  for (let lv = 1; lv <= 3; lv++) {
    const rr = r * lv / 3;
    const pts = [];
    for (let i = 0; i < n; i++) {
      const angle = (Math.PI * 2 * i / n) - Math.PI / 2;
      pts.push(`${cx + rr * Math.cos(angle)},${cy + rr * Math.sin(angle)}`);
    }
    svg += `<polygon points="${pts.join(' ')}" fill="none" stroke="#E0E0E0" stroke-width="${lv===3?1.5:0.7}"/>`;
  }

  for (let i = 0; i < n; i++) {
    const angle = (Math.PI * 2 * i / n) - Math.PI / 2;
    const x2 = cx + r * Math.cos(angle);
    const y2 = cy + r * Math.sin(angle);
    svg += `<line x1="${cx}" y1="${cy}" x2="${x2}" y2="${y2}" stroke="#E0E0E0" stroke-width="0.7"/>`;
  }

  const dataPts = [];
  for (let i = 0; i < n; i++) {
    const val = Math.min(scores[i] || 0, 1);
    const angle = (Math.PI * 2 * i / n) - Math.PI / 2;
    const dr = r * (0.1 + 0.9 * val);
    dataPts.push(`${cx + dr * Math.cos(angle)},${cy + dr * Math.sin(angle)}`);
  }
  const hasData = scores.some(s => s > 0);
  if (hasData) {
    svg += `<polygon points="${dataPts.join(' ')}" fill="rgba(255,107,53,0.15)" stroke="var(--primary)" stroke-width="2" stroke-linejoin="round"${animate ? ' style="animation:fadeUp 0.5s ease"' : ''}/>`;
    for (let i = 0; i < n; i++) {
      if ((scores[i] || 0) > 0) {
        svg += `<circle cx="${dataPts[i].split(',')[0]}" cy="${dataPts[i].split(',')[1]}" r="3.5" fill="var(--primary)"/>`;
      }
    }
  }

  if (showLabels) {
    for (let i = 0; i < n; i++) {
      const angle = (Math.PI * 2 * i / n) - Math.PI / 2;
      const lr = r + size * 0.12;
      const lx = cx + lr * Math.cos(angle);
      const ly = cy + lr * Math.sin(angle);
      const val = scores[i] || 0;
      const color = val > 0 ? AXIS_COLORS[i] : '#AAA';
      const fontSize = size > 100 ? 12 : 6;
      svg += `<text x="${lx}" y="${ly}" text-anchor="middle" dominant-baseline="central" fill="${color}" font-size="${fontSize}" font-weight="700">${AXES[i]}</text>`;
      if (showValues && val === 0) {
        const vly = ly + fontSize + 2;
        svg += `<text x="${lx}" y="${vly}" text-anchor="middle" fill="#CCC" font-size="${fontSize * 0.7}">未冒険</text>`;
      }
    }
  }

  svgEl.innerHTML = svg;
}

/* ===== ユーザーのスコア計算 ===== */
function calcScores(user) {
  const scores = [0,0,0,0,0,0];
  const records = user.records;
  const count = Object.keys(records).length;
  if (count === 0) return scores;

  for (const boothId in records) {
    const booth = BOOTHS.find(b => b.id === parseInt(boothId));
    if (!booth) continue;
    const rec = records[boothId];
    let weight = 1;
    if (rec.q1 === 'fun') weight = 1.0;
    else if (rec.q1 === 'wow') weight = 1.2;
    else if (rec.q1 === 'curious') weight = 1.1;
    let diffBonus = 0;
    if (rec.q2 === 'hard') diffBonus = 0.15;
    else if (rec.q2 === 'normal') diffBonus = 0.05;

    for (let i = 0; i < 6; i++) {
      scores[i] += booth.axes[i] * weight + (booth.axes[i] > 0 ? diffBonus : 0);
    }
  }

  const max = Math.max(...scores, 0.001);
  return scores.map(s => Math.min(s / max, 1));
}

function renderReview() {
  const user = getCurrentUser(appState);
  const scores = calcScores(user);
  const displayName = user.name || 'ゲスト';
  const records = user.records;

  document.getElementById('reviewUserName').textContent = `${displayName}のワクワクマップ`;
  drawRadar(document.getElementById('reviewChart'), scores, 260, { showLabels: true, showValues: true, animate: true });

  // === タイプ判定 ===
  let maxIdx = 0;
  for (let i = 1; i < scores.length; i++) {
    if (scores[i] > scores[maxIdx]) maxIdx = i;
  }
  const topAxis = AXES[maxIdx];
  const typeName = TYPE_NAMES[topAxis];
  document.getElementById('typeName').textContent = `${topAxis}タイプ — ${typeName}`;

  // === 体験パターン抽象化メッセージ（v3新規） ===
  // 記録された感情×ブース軸の最頻パターンを特定し、動的に説明文を生成
  const patternCounts = {};
  for (const boothId in records) {
    const booth = BOOTHS.find(b => b.id === parseInt(boothId));
    if (!booth) continue;
    const rec = records[boothId];
    const emotion = rec.q1; // fun / wow / curious
    // このブースで最も強い軸を特定
    let strongestAxisIdx = 0;
    let strongestVal = 0;
    for (let i = 0; i < 6; i++) {
      if (booth.axes[i] > strongestVal) {
        strongestVal = booth.axes[i];
        strongestAxisIdx = i;
      }
    }
    const key = `${emotion}×${AXES[strongestAxisIdx]}`;
    patternCounts[key] = (patternCounts[key] || 0) + 1;
  }

  // 最頻パターンを取得
  let dominantPattern = '';
  let dominantCount = 0;
  for (const key in patternCounts) {
    if (patternCounts[key] > dominantCount) {
      dominantCount = patternCounts[key];
      dominantPattern = key;
    }
  }

  const patternText = EMOTION_AXIS_PATTERNS[dominantPattern] || `いろんな分野にバランスよく興味を持てるタイプ`;
  document.getElementById('typeDesc').textContent = `きみは${patternText}！`;

  // === 体験からの気づき（v3新規：具体的なブース名を使った抽象化） ===
  const insightCard = document.getElementById('insightCard');
  const insightText = document.getElementById('insightText');
  const recordedBooths = Object.keys(records).map(id => {
    const booth = BOOTHS.find(b => b.id === parseInt(id));
    return { booth, rec: records[id] };
  }).filter(r => r.booth);

  if (recordedBooths.length >= 2) {
    // 感情が同じブースをグルーピングして共通点を見つける
    const byEmotion = { fun: [], wow: [], curious: [] };
    recordedBooths.forEach(r => {
      if (byEmotion[r.rec.q1]) byEmotion[r.rec.q1].push(r.booth);
    });

    let insightMsg = '';
    const emotionLabels = { fun: '「たのしい！」', wow: '「すごい！」', curious: '「もっと知りたい！」' };

    // 同じ感情で2件以上あればパターンを指摘
    for (const emo in byEmotion) {
      if (byEmotion[emo].length >= 2) {
        const names = byEmotion[emo].map(b => b.name).join('と');
        insightMsg = `${names}で${emotionLabels[emo]}と感じたね。`;
        // 共通する強い軸を見つける
        const axisTotals = [0,0,0,0,0,0];
        byEmotion[emo].forEach(b => {
          for (let i = 0; i < 6; i++) axisTotals[i] += b.axes[i];
        });
        let commonAxisIdx = 0;
        for (let i = 1; i < 6; i++) {
          if (axisTotals[i] > axisTotals[commonAxisIdx]) commonAxisIdx = i;
        }
        insightMsg += `どちらも「${AXES[commonAxisIdx]}」の要素が強い体験だよ。きみは${AXES[commonAxisIdx]}の分野で力を発揮できそう！`;
        break;
      }
    }

    // 同じ感情が2件ない場合は、全体の傾向で述べる
    if (!insightMsg) {
      const boothNames = recordedBooths.map(r => r.booth.name);
      const emotions = recordedBooths.map(r => emotionLabels[r.rec.q1]);
      insightMsg = `${boothNames[0]}では${emotions[0]}、${boothNames[1]}では${emotions[1]}と感じたね。`;
      insightMsg += `いろんな感じ方ができるのがきみの強み。いろんな体験に飛び込んでみよう！`;
    }

    insightText.textContent = insightMsg;
    insightCard.style.display = 'block';
  } else {
    insightCard.style.display = 'none';
  }

  // === おすすめブース＋「観点」付与（v3改修） ===
  const unvisited = BOOTHS.filter(b => !records[b.id]);
  // 最も弱い軸（未冒険 or 最低スコア）を特定
  let weakestAxisIdx = 0;
  let weakestScore = Infinity;
  for (let i = 0; i < scores.length; i++) {
    if (scores[i] < weakestScore) {
      weakestScore = scores[i];
      weakestAxisIdx = i;
    }
  }
  const weakestAxis = AXES[weakestAxisIdx];

  const ranked = unvisited.map(b => {
    let affinity = 0;
    for (let i = 0; i < 6; i++) affinity += b.axes[i] * (1 - (scores[i] || 0));
    return { booth: b, affinity };
  }).sort((a, b) => b.affinity - a.affinity);

  const recommendList = document.getElementById('recommendList');
  const recommendSection = document.getElementById('recommendSection');
  if (ranked.length === 0) {
    recommendSection.style.display = 'none';
  } else {
    recommendSection.style.display = 'block';
    recommendList.innerHTML = '';
    ranked.slice(0, 2).forEach(r => {
      // このブースの最強軸を特定
      let boothStrongIdx = 0;
      let boothStrongVal = 0;
      for (let i = 0; i < 6; i++) {
        if (r.booth.axes[i] > boothStrongVal) {
          boothStrongVal = r.booth.axes[i];
          boothStrongIdx = i;
        }
      }
      const boothStrongAxis = AXES[boothStrongIdx];

      // 「なぜこの観点で行くべきか」の理由文を生成
      const relationKey = `${topAxis}→${boothStrongAxis}`;
      const perspective = AXIS_RELATIONS[relationKey] || `「${boothStrongAxis}」の視点で新しい発見があるよ`;

      const card = document.createElement('div');
      card.className = 'recommend-card';
      card.innerHTML = `
        <div class="booth-icon" style="background:${r.booth.color}">${r.booth.icon}</div>
        <div class="recommend-card-info">
          <div class="recommend-card-name">${r.booth.name}</div>
          <div class="recommend-card-reason">きみの「${topAxis}」体験が多いから、次は「${boothStrongAxis}」の視点で見てみよう</div>
          <div class="recommend-card-perspective">→ ${perspective}</div>
        </div>
      `;
      recommendList.appendChild(card);
    });
  }

  // === 日常アクション（v3改修：afterActionsから感情別に選出） ===
  const actionList = document.getElementById('actionList');
  actionList.innerHTML = '';
  const actions = [];

  for (const boothId in records) {
    const booth = BOOTHS.find(b => b.id === parseInt(boothId));
    if (!booth) continue;
    const rec = records[boothId];
    const emotion = rec.q1; // fun / wow / curious
    const action = booth.afterActions[emotion];
    if (action) {
      actions.push({ text: action, boothName: booth.name, emotion });
    }
  }

  // 最大3つ表示（記録数が多い場合は、感情の多様性を優先して選出）
  const selectedActions = [];
  const usedEmotions = new Set();
  // まず各感情から1つずつ選ぶ（多様性確保）
  for (const a of actions) {
    if (!usedEmotions.has(a.emotion) && selectedActions.length < 3) {
      selectedActions.push(a);
      usedEmotions.add(a.emotion);
    }
  }
  // 3つ未満なら残りから補填
  for (const a of actions) {
    if (selectedActions.length >= 3) break;
    if (!selectedActions.includes(a)) {
      selectedActions.push(a);
    }
  }

  selectedActions.forEach((a, i) => {
    const card = document.createElement('div');
    card.className = 'action-card';
    card.innerHTML = `
      <div class="action-num">${i + 1}</div>
      <div>
        <div class="action-text">${a.text}</div>
        <div class="action-source">← ${a.boothName}の体験から</div>
      </div>
    `;
    actionList.appendChild(card);
  });
}

**v2→v3の変更点（ふりかえり画面のデータ層改修のみ、UIレイアウト変更なし）：**

### 1. ブースデータに`afterActions`を追加
- 8社×感情3種＝24パターンの日常アクション文言を手書きで追加
- 例：トラック体験で「たのしかった！」→「家にある材料でミニカーや乗り物を工作してみよう」、「すごかった！」→「トラックがどうやって動くか、図鑑やネットで仕組みを調べてみよう」
- 「明日からやってみよう」は固定文言を廃止し、**実際に記録したブース×感情回答の組み合わせ**から選出。感情の多様性を優先してバランスよく3つ選ぶ
- 各アクションに「← トラックづくり体験の体験から」と出典を表示

### 2. おすすめブースに「観点」を付与
- `AXIS_RELATIONS`（30パターンの軸間関係データ）を追加
- おすすめブースに「きみの"つくる"体験が多いから、次は"まもる"の視点で見てみよう」＋具体的な理由文を表示
- 例：つくる→まもる「作ったものを"守る"側の視点を知ると、ものづくりがもっと深くなるよ」

### 3. 体験パターンの抽象化メッセージを追加
- `EMOTION_AXIS_PATTERNS`（感情×軸の18パターン）から動的に傾向文を生成
- 固定の`type.desc`を廃止し、実際の記録パターンに基づく文に差し替え
- 新設の「体験からの気づき」カードで、具体的なブース名を使った抽象化を表示
  - 例：「トラックづくり体験とパンづくり体験で"たのしい！"と感じたね。どちらも"つくる"の要素が強い体験だよ。きみはつくるの分野で力を発揮できそう！」

**検証：「トラック＋パン」と「消防士＋指紋採取」で異なる出力が出ることを確認済み**
- トラック(fun)＋パン(fun) → つくるタイプ・クリエイター、「手を動かしてモノを作る体験がいちばん楽しい」、日常アクションは「ミニカー工作」「パン作り」
- 消防士(wow)＋指紋採取(curious) → まもるタイプ・ガーディアン、「人を守る仕事の迫力やかっこよさに心を動かされる」、日常アクションは「消防車の種類を調べる」「家の中で証拠探しゲーム」
